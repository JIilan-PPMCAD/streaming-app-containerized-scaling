pipeline {
    agent any
    environment {
        // Your verified AWS Account and Regional Details
        AWS_ACCOUNT_ID       = '319918321628'
        AWS_DEFAULT_REGION   = 'ap-south-1'
        CLUSTER_NAME         = 'mern-production-cluster'
        SNS_TOPIC_ARN        = 'arn:aws:sns:ap-south-1:319918321628:mern-deployment-alerts'
        
        // Exact names matching your verified AWS ECR Console configuration strings
        FRONTEND_REPO        = 'mern-frontend'
        HELLO_SERVICE_REPO   = 'mern-hello-service'
        PROFILE_SERVICE_REPO = 'mern-profile-service'
        
        IMAGE_TAG            = "${BUILD_NUMBER}"
    }
    stages {
        stage('Checkout Source') {
            steps {
                cleanWs()
                checkout scm
            }
        }
        stage('AWS ECR Authentication') {
            steps {
                withCredentials([[$class: 'AmazonWebServicesCredentialsBinding', credentialsId: 'aws-ecr-creds']]) {
                    sh """
                    aws ecr get-login-password --region ${AWS_DEFAULT_REGION} | \
                    docker login --username AWS --password-stdin ${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_DEFAULT_REGION}.amazonaws.com
                    """
                }
            }
        }
        stage('Parallel Multi-Service Build & Push') {
            parallel {
                stage('Frontend Service') {
                    steps {
                        script {
                            dir('frontend') {
                                // FIXED: Replaced invalid '://{FRONTEND_REPO}' with valid AWS ECR registry path domain layout
                                def frontendImg = docker.build("${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_DEFAULT_REGION}://{FRONTEND_REPO}:${IMAGE_TAG}")
                                frontendImg.push()
                                frontendImg.push("latest")
                            }
                        }
                    }
                }
                stage('Hello Service (Backend)') {
                    steps {
                        script {
                            dir('backend/helloService') {
                                // FIXED: Replaced invalid '://{HELLO_SERVICE_REPO}' with valid AWS ECR registry path domain layout
                                def helloImg = docker.build("${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_DEFAULT_REGION}://{HELLO_SERVICE_REPO}:${IMAGE_TAG}")
                                helloImg.push()
                                helloImg.push("latest")
                            }
                        }
                    }
                }
                stage('Profile Service (Backend)') {
                    steps {
                        script {
                            dir('backend/profileService') {
                                // FIXED: Replaced invalid '://{PROFILE_SERVICE_REPO}' with valid AWS ECR registry path domain layout
                                def profileImg = docker.build("${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_DEFAULT_REGION}://{PROFILE_SERVICE_REPO}:${IMAGE_TAG}")
                                profileImg.push()
                                profileImg.push("latest")
                            }
                        }
                    }
                }
            }
        }
        stage('EKS Kubernetes Deployment (Helm)') {
            steps {
                withCredentials([[$class: 'AmazonWebServicesCredentialsBinding', credentialsId: 'aws-ecr-creds']]) {
                    sh """
                    # 1. Update kubeconfig to connect Jenkins directly to your live Mumbai EKS cluster
                    aws eks update-kubeconfig --region ${AWS_DEFAULT_REGION} --name ${CLUSTER_NAME}
                    
                    # 2. Deploy all 3 services via Helm, passing the exact build numbers to trigger rolled updates
                    helm upgrade --install mern-production ./charts/mern-stack \
                      --set frontend.tag=${IMAGE_TAG} \
                      --set helloService.tag=${IMAGE_TAG} \
                      --set profileService.tag=${IMAGE_TAG} \
                      --values ./charts/mern-stack/values.yaml
                    """
                }
            }
        }
        stage('Post-Deployment Health Validation') {
            steps {
                sh """
                echo "=== Verifying Kubernetes Pod Deployment States ==="
                kubectl get pods -n default
                
                echo "=== Retrieving Live Public Facing Frontend Ingress URL ==="
                kubectl get svc ${FRONTEND_REPO}-svc
                echo ""
                """
            }
        }
    }
    post {
        success {
            withCredentials([[$class: 'AmazonWebServicesCredentialsBinding', credentialsId: 'aws-ecr-creds']]) {
                sh "aws sns publish --topic-arn ${SNS_TOPIC_ARN} --region ${AWS_DEFAULT_REGION} --message '✅ Pipeline Success: Build #${BUILD_NUMBER} has been successfully deployed to EKS cluster!'"
            }
        }
        failure {
            withCredentials([[$class: 'AmazonWebServicesCredentialsBinding', credentialsId: 'aws-ecr-creds']]) {
                sh "aws sns publish --topic-arn ${SNS_TOPIC_ARN} --region ${AWS_DEFAULT_REGION} --message '🚨 Pipeline Failure: Build #${BUILD_NUMBER} failed during execution. Please check the Jenkins console logs immediately.'"
            }
        }
        always {
            // Cleans up old dangling build layers to preserve EC2 disk storage capacity
            sh "docker system prune -f"
        }
    }
}
